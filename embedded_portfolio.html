ring_buffer.h C++17 
// Thread-safe ring buffer for embedded UART DMA — lock-free with atomics
#pragma once
#include <atomic>
#include <cstdint>
#include <optional>

template <typename T, size_t N>
class RingBuffer {
    static_assert((N & (N - 1)) == 0, "N must be a power of 2");
    static constexpr size_t MASK = N - 1;

    T buf_[N];
    alignas(64) std::atomic<size_t> head_{0};
    alignas(64) std::atomic<size_t> tail_{0};

public:
    // ISR-safe push (producer side)
    bool push(const T& val) noexcept {
        size_t h = head_.load(std::memory_order_relaxed);
        size_t next = (h + 1) & MASK;
        if (next == tail_.load(std::memory_order_acquire))
            return false; // full
        buf_[h] = val;
        head_.store(next, std::memory_order_release);
        return true;
    }

    // Task-side pop returning std::optional
    std::optional<T> pop() noexcept {
        size_t t = tail_.load(std::memory_order_relaxed);
        if (t == head_.load(std::memory_order_acquire))
            return std::nullopt; // empty
        T val = buf_[t];
        tail_.store((t + 1) & MASK, std::memory_order_release);
        return val;
    }

    bool   empty() const noexcept { return head_ == tail_; }
    size_t capacity() const noexcept { return N - 1; }
};
uart_driver.c C99 
#include "uart_driver.h"
#include "ring_buffer.h"
#define UART_BAUD_115200  0x8A    /* baud divisor register value  */
#define UART_FIFO_DEPTH   64

static volatile uint8_t rx_buf[UART_FIFO_DEPTH];
static volatile uint32_t rx_head, rx_tail;

/* Interrupt Service Routine — runs in privileged mode */
void __attribute__((interrupt("IRQ"))) UART0_IRQHandler(void) {
    uint8_t status = MMIO_READ8(UART0_LSR);

    while (status & LSR_DATA_READY) {
        uint8_t byte = MMIO_READ8(UART0_RBR);
        uint32_t next = (rx_head + 1) % UART_FIFO_DEPTH;
        if (next != rx_tail) {        /* drop on overflow */
            rx_buf[rx_head] = byte;
            rx_head = next;
        }
        status = MMIO_READ8(UART0_LSR);
    }
    MMIO_WRITE8(UART0_EOI, 0x01);  /* clear interrupt */
}

int32_t uart_read(uint8_t *out, uint32_t len) {
    uint32_t read = 0;
    while (read < len && rx_tail != rx_head) {
        out[read++] = rx_buf[rx_tail];
        rx_tail = (rx_tail + 1) % UART_FIFO_DEPTH;
    }
    return (int32_t)read;
}

#!/usr/bin/env bash
set -euo pipefail

TARGET="stm32h7"
CONFIG="debug"
DEVICE="/dev/ttyUSB0"

log() { echo "[$(echo $1 | tr '[:lower:]' '[:upper:]')] ${@:2}"; }
die() { log ERROR "$*"; exit 1; }

# Parse CLI args
while [[ $# -gt 0 ]]; do
  case $1 in
    --target) TARGET="$2"; shift 2 ;;
    --config) CONFIG="$2"; shift 2 ;;
    *)        die "Unknown arg: $1" ;;
  esac
done

# Toolchain check
command -v arm-none-eabi-gcc &>/dev/null || die "Toolchain not found"
VER=$(arm-none-eabi-gcc --version | head -1)
log build "Checking toolchain: $VER ✓"

# CMake build
mkdir -p build
pushd build >/dev/null
cmake .. -DCMAKE_BUILD_TYPE="${CONFIG^}" -DTARGET="$TARGET" -GNinja
ninja -j$(nproc)
popd >/dev/null

# Flash via OpenOCD
openocd -f interface/stlink.cfg -f target/"$TARGET".cfg \
  -c "program build/firmware.elf verify reset exit"

spi_driver.c — reading timing from datasheet C 
/* SPI timing constants derived from device datasheet Table 12 */
#define SPI_CLK_HZ        10000000UL   /* 10 MHz max SCLK            */
#define SPI_SETUP_NS      12           /* tSU: data setup before CLK */
#define SPI_HOLD_NS       8            /* tHD: data hold after CLK   */
#define SPI_CS_ASSERT_NS  20           /* tCSS: CS low before SCLK   */

uint8_t spi_transfer(uint8_t tx) {
    GPIO_LOW(SPI_CS_PIN);
    delay_ns(SPI_CS_ASSERT_NS);

    uint8_t rx = 0;
    for (int bit = 7; bit >= 0; bit--) {
        GPIO_WRITE(SPI_MOSI_PIN, (tx >> bit) & 1);
        delay_ns(SPI_SETUP_NS);
        GPIO_HIGH(SPI_CLK_PIN);
        rx = (rx << 1) | GPIO_READ(SPI_MISO_PIN);
        delay_ns(SPI_HOLD_NS);
        GPIO_LOW(SPI_CLK_PIN);
    }

    GPIO_HIGH(SPI_CS_PIN);
    return rx;
}

import pytest
from uart_sim import UARTDevice, loopback_fixture

class TestUARTProtocol:

    def test_hello_handshake(self, loopback_fixture):
        """Device responds with ACK within 5ms of ENQ."""
        dev = loopback_fixture
        dev.send_byte(0x05)  # ENQ
        resp = dev.wait_byte(timeout_ms=5)
        assert resp == 0x06, f"Expected ACK(0x06), got 0x{resp:02X}"

    def test_baud_tolerance(self, loopback_fixture):
        """Device accepts ±2% baud rate deviation per datasheet."""
        for baud in [112700, 115200, 117500]:
            loopback_fixture.set_baud(baud)
            assert loopback_fixture.ping() == True

    @pytest.mark.parametrize("payload,expected_crc", [
        (b"\x01\x02\x03", 0xD2),
        (b"\xFF\x00\x80", 0x1A),
    ])
    def test_crc8_checksum(self, payload, expected_crc):
        from uart_sim import crc8_maxim
        assert crc8_maxim(payload) == expected_crc
